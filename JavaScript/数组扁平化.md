# 数组扁平化

@

- 原生数组方法

  ```javascript
  function fn1(arr = []) {
        return arr.flat(Infinity)
  }
  ```

- 递归

  ```javascript
  function fn2(arr = []) {
        return arr.reduce((pre, cur) => {
              if (Array.isArray(cur)) {
                    pre.push(...fn2(cur))
              } else {
                    pre.push(cur);
              }
              return pre;
        }, [])
  }
  ```

- 迭代器

  ```javascript
  function fn3(arr = []) {
        function* iteration(arr) {
              for (let item of arr) {
                    if (Array.isArray(item)) {
                        	// 转移迭代器控制权
                          yield* iteration(item)
                    } else {
                          yield item
                    }
              }
        }
  
        return [...iteration(arr)];
  }
  ```

- 栈存取

  ```javascript
  function fn4(arr = []) {
        // 避免修改原数组
        arr = arr.slice()
        const res = [];
        while (arr.length) {
              // 获取头部
              const value = arr.shift();
              // 是数组的话解构放回头部
              if (Array.isArray(value)) {
                    arr.unshift(...value)
              } else {
                    res.push(value)
              }
        }
        return res;
  }
  ```

- 遍历

  第四个方法的优化版本，相比方法四，考虑不改变原数组的情况下，也只需要一个额外数组

  ```javascript
  function fn5(arr = []) {
        // 避免修改原数组
        const res = arr.slice()
        for (let i = 0; i < res.length; i++) {
              const value = res[i];
              if (Array.isArray(value)) {
                    // 将数据解构插入原来元素的位置
                    res.splice(i, 1, ...value);
                    // 修正指针位置
                    i--;
              }
        }
        return res;
  }
  ```

### 测试

```javascript
let arr = [1, [2, 3, [4], 5], 6]

console.log(fn1(arr)); // [1, 2, 3, 4, 5, 6]
console.log(fn2(arr)); // [1, 2, 3, 4, 5, 6]
console.log(fn3(arr)); // [1, 2, 3, 4, 5, 6]
console.log(fn4(arr)); // [1, 2, 3, 4, 5, 6]
console.log(fn5(arr)); // [1, 2, 3, 4, 5, 6]
```
