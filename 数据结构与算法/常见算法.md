# 收录一些需要掌握的常见算法

## 斐波那契数列

```js
function fibonacci(n) {
  if (n < 1) return 0
  if (n <= 2) return 1
  return fibonacci(n - 1) + fibonacci(n - 2)
}
```

## 合并两个有序数组

```js
function mergeArray(arrA, arrB) {
  let i = 0
  let j = 0
  let result = []
  while (i < arrA.length && j < arrB.length) {
    if (arrA[i] < arrB[j]) {
      result.push(arrA[i++])
    } else {
      result.push(arrB[j++])
    }
  }
  while (i < arrA.length) {
    result.push(arrA[i++])
  }
  while (j < arrB.length) {
    result.push(arrB[j++])
  }
  return result
}
```

## 窗口

```js
// 求字符串中不重复子字符串的最大长度
// 用移动窗口的思想解答
// 每次窗口向右增加一格，如果存在重复值，则移除窗口最左边直到当前窗口不存在重复值
// 最后只需返回所存的最大值即可
function a(str) {
  if (str.length === 0) return 0
  let left = 0 // 记录窗口最左边相对于str的下标
  let result = 1
  let cur = str[0]
  for (let i = 1; i < str.length; i++) {
    const nextCode = str[i]
    while (cur.includes(nextCode)) {
      left++
      cur = str.slice(left, i)
    }
    cur = cur + nextCode
    result = result > cur.length ? result : cur.length
  }
  return result
}
```
